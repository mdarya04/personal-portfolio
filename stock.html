For real-time stock market tracking, you'll need a backend service to fetch and serve the stock data. You can use Flask for this purpose..

Create a Flask App (app.py): 

                        import yfinance as yf
                        from flask import Flask, jsonify, request
                        
                        app = Flask(__name__)
                        
                        @app.route('/api/stock/', methods=['GET'])
                        def get_stock(ticker):
                            try:
                                stock = yf.Ticker(ticker)
                                data = stock.history(period='1d', interval='1m')
                                latest_data = data.iloc[-1]  # Get the latest data point
                                result = {
                                    'time': latest_data.name.strftime('%Y-%m-%d %H:%M:%S'),
                                    'open': latest_data['Open'],
                                    'high': latest_data['High'],
                                    'low': latest_data['Low'],
                                    'close': latest_data['Close'],
                                    'volume': latest_data['Volume']
                                }
                                return jsonify(result)
                            except Exception as e:
                                return jsonify({'error': str(e)}), 500
                        
                        if __name__ == '__main__':
                            app.run(debug=True)

Stock Market Tracker Code

This section contains the frontend code for the stock market tracker application.

stock_tracker.py 

                        import yfinance as yf
                        import pandas as pd
                        import time
                        
                        def fetch_stock_data(ticker):
                            """
                            Fetches real-time stock data for a given ticker symbol.
                            """
                            stock = yf.Ticker(ticker)
                            data = stock.history(period='1d', interval='1m')
                            return data
                        
                        def display_stock_data(ticker):
                            """
                            Continuously fetches and displays real-time stock data for a given ticker symbol.
                            """
                            print(f"Real-Time Stock Market Tracker for {ticker}")
                            print(f"{'Time':<20} {'Open':<10} {'High':<10} {'Low':<10} {'Close':<10} {'Volume':<10}")
                            
                            while True:
                                try:
                                    data = fetch_stock_data(ticker)
                                    latest_data = data.iloc[-1]
                                    current_time = pd.to_datetime(latest_data.name).strftime('%Y-%m-%d %H:%M:%S')
                                    print(f"{current_time:<20} {latest_data['Open']:<10.2f} {latest_data['High']:<10.2f} {latest_data['Low']:<10.2f} {latest_data['Close']:<10.2f} {latest_data['Volume']:<10}")
                                    
                                    # Wait for 60 seconds before fetching data again
                                    time.sleep(60)
                                except KeyboardInterrupt:
                                    print("\nReal-Time Stock Market Tracker stopped.")
                                    break
                                except Exception as e:
                                    print(f"An error occurred: {e}")
                                    time.sleep(60)
                        
                        if __name__ == "__main__":
                            ticker = input("Enter the stock ticker symbol (e.g., AAPL, TSLA, MSFT): ").upper()
                            display_stock_data(ticker)
